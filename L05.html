Functions with variables/parameters/arguments:

function drawEye(x,y, r,g,b, eyeSize) {

  fill(255); // eye white
  ellipse(x, y, eyeSize, eyeSize);

  fill(r, g, b); // green eye color
  ellipse(x, y, eyeSize/2, eyeSize/2);

  fill(0); // black pupil
  ellipse(x, y eyeSize/4, eyeSize/4);
}
